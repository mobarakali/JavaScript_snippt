Working With JavaScript Loops

While Loop

   // While loop
    var n = 1;
    while (n<=5){
        console.log( "While Loop # "+ n++);
    }

Do While Loop

   // Do While loop
   n = 1;
   do{
       console.log( "Do While loop # "+ n++);
   }while (n<=5);

For Loop

// For loop
    for(n = 1;n<=5; n++){
        console.log( "For loop # "+ n);
    }
