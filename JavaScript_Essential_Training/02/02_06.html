JavaScript Operator

Working With JavaScript Operators

     var a = 1;
    var b = "1";
    // First Condition
    if(a==b){
        alert("(a == b): a equals b.");
    }
    // Another condition
    if(a===b){
        alert("(a === b): a equals b.");
    }else{
        alert("(a === b): a not-equals b.");
    }

    // Using And (&&)

    if(a==b && a===b){
        alert( "a & b are very much similar!")
    }
    // or we can write
    if( (a==b) && (a===b) ){
        alert( "a & b are very much similar!")
    }

    // Again we can write
    if(
            (a==b)
            &&
            (a===b)
    ){
        alert( "a & b are very much similar!")
    }

    // using or (||)
    if( (a==b) || (a===b) ){
        alert( "a & b are very much similar!")
    }
